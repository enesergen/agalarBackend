CREATE TABLE News (
    ID SERIAL PRIMARY KEY,
    header varchar(255) NOT NULL,
    short varchar(255),
    body varchar(5000)
);
CREATE TABLE Activities (
    ID SERIAL PRIMARY KEY,
    header varchar(255) NOT NULL,
    short varchar(255),
    body varchar(5000)
);
CREATE TABLE Administration (
    ID SERIAL PRIMARY KEY,
    Degree varchar(255) NOT NULL,
    name varchar(255),
    body varchar(5000)
);
CREATE TABLE AdministrativeUnits (
    ID SERIAL PRIMARY KEY,
    name varchar(255),
    body varchar(5000)
);
CREATE TABLE lecturers (
    ID SERIAL PRIMARY KEY,
	Degree varchar(255) NOT NULL,
    name varchar(255),
    body varchar(5000)
);

CREATE TABLE canteen (
    ID SERIAL PRIMARY KEY,
	brand varchar(255) NOT NULL,
    name varchar(255),
    price float
);
CREATE TABLE calendar (
    ID SERIAL PRIMARY KEY,
    name varchar(255),
	description varchar(255),
    date date not null
);

CREATE TABLE campuses (
    ID SERIAL PRIMARY KEY,
    name varchar(255),
	location varchar(255)
);
CREATE TABLE faculty (
    ID SERIAL PRIMARY KEY,
	CID INT,
    name varchar(255),
	location varchar(255)
);

CREATE TABLE departments (
    ID SERIAL PRIMARY KEY,
	FID INT,
    name varchar(255),
	location varchar(255)
);

CREATE TABLE slides (
    ID SERIAL PRIMARY KEY,
	URL varchar(255),
	slideorder integer
);
CREATE TABLE facultyslides (
    ID SERIAL PRIMARY KEY,
	fid integer,
    url varchar(255),
	slideorder integer
);
